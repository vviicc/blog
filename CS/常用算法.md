###常用算法

#####二分查找算法：

    int BinarySearch(int s[],int length,int key)
    {
      int low,high,middle;
      low = 0;
      high= length - 1;
      while(low <= high){
        middle = low + (high-low)>>1;
        if(s[middle] == key)
            return middle;
        else if(s[middle] > key)
            high = middle - 1;
        else
            low = middle + 1;
      }
      return -1;
    }
    
    
#####快速排序算法：
	void quickSort(int s[],int left,int right)
	{
		if(left < right){
			int l = left;
			int r = right;
			int x = s[left];
			
			while(l < r){
				while(l < r && s[r] >= x)	j--;
				if(l < r) s[i++] = s[j];
				while(l < r && s[l] <= x)	i++;
				if(l < r) s[j--] = s[i];
			}
			s[i] = x;
			quickSort(s,l,i-1);
			quickSort(s,i+1,r);
		}
	}
